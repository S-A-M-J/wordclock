cd $HOME/.node-red
npm i node-red-dashboard

cd

touch node
sed -i "server {listen 80;
	server_name wordclock.de;
		location /ui {
		proxy_set_header X-Forwarded-For $remote_addr;
		proxy_set_header Host $http_host;
		proxy_pass "http://127.0.0.1:1880";
		}
	}" node
